Python: Printing a class/object

(First posted on: 2009-10-08 06:58:30+00:00)

class Foobar:

    def __init__(self, val0, val1):
        self.val0 = val0
        self.val1 = val1

    def __repr__(self):
        """String representation of class/object."""
        return "{} and {}".format(self.val0, self.val1)

fbar = Foobar(99, 100)
print(fbar) # 99 and 100
