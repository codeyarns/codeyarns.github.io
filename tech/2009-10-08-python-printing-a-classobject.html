class Foobar: def init(self, val0, val1): self.val0 = val0 self.val1 = val1 def repr(self): """String representation of class/object.""" return "{} and {}".format(self.val0, self.val1)

fbar = Foobar(99, 100)
print(fbar) # 99 and 100
